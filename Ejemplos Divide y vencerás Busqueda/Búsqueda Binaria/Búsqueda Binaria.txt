public class BusquedaBinaria {
    public static void main(String[] args) {
        // Arreglo ORDENADO
        int[] numeros = {2, 4, 6, 8, 10, 12, 14};
        int buscado = 10;
        
        int resultado = busquedaBinaria(numeros, buscado);
        
        if (resultado == -1) {
            System.out.println("El número " + buscado + " no se encuentra en el arreglo.");
        } else {
            System.out.println("El número " + buscado + " se encuentra en la posición: " + resultado);
        }
    }
    
    // Método para realizar la búsqueda binaria
    public static int busquedaBinaria(int[] arreglo, int valor) {
        int inicio = 0;
        int fin = arreglo.length - 1;
        
        while (inicio <= fin) {
            int medio = (inicio + fin) / 2;
            
            if (arreglo[medio] == valor) {
                return medio; // Encontrado
            } else if (arreglo[medio] < valor) {
                inicio = medio + 1; // Buscar en la mitad derecha
            } else {
                fin = medio - 1; // Buscar en la mitad izquierda
            }
        }
        return -1;
    }
}
